#include <stdio.h>

void main()
{
    int x = 0, a = 0, b = 0, c = 0, result = 0;
    scanf("%d", &x);
    if (x > 999 | x < 0) {
        printf("Error");
    } else {
        a = x / 100;
        b = (x - a * 100) / 10;
        c = x - a * 100 - b * 10;
        result = a + b + c;
        if (x % 3 == 0 & x % 2 == 1) {
            printf("특별한 숫자");
        } else if (result % 7 == 0) {
            printf("특별한 숫자");
        } else if (a > c) {
            printf("특별한 숫자");
        } else {
            printf("특별한 숫자가 아닙니다.");
        }
    }
}
