#include <stdio.h>

void main()
{
    int x = 0;
    char y;
    scanf("%d", &x);
    switch(x)
    {
        case 0:
            y = 'A';
            printf("%d → %c", x, y);
            break;
        case 1:
            y = 'B';
            printf("%d → %c", x, y);
            break;
        case 2:
            y = 'C';
            printf("%d → %c", x, y);
            break;
        case 3:
            y = 'D';
            printf("%d → %c", x, y);
            break;
        case 4:
            y = 'E';
            printf("%d → %c", x, y);
            break;
        case 5:
            y = 'F';
            printf("%d → %c", x, y);
            break;
        case 6:
            y = 'G';
            printf("%d → %c", x, y);
            break;
        case 7:
            y = 'H';
            printf("%d → %c", x, y);
            break;
        case 8:
            y = 'I';
            printf("%d → %c", x, y);
            break;
        case 9:
            y = 'J';
            printf("%d → %c", x, y);
            break;
        default:
            printf("잘못된 입력입니다!");
    }
}
