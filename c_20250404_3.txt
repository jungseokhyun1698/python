#include <stdio.h>

void main()
{
    int score = 0;
    float bonus = 0;
    printf("시험 점수를 입력하세요.\n");
    scanf("%d", &score);
    int grade = score / 10;
    if (score > 100 | score < 0) {
        printf("Error");
    } else {
        switch(grade)
        {
            case 10:
            case 9:
                printf("A 등급\n");
                bonus = score * 0.1;
                printf("%4.2f점 추가 보너스\n", bonus);
                printf("보너스 포함 %4.2f점", score + bonus);
                break;
            case 8:
                printf("B 등급\n");
                bonus = score * 0.05;
                printf("%4.2f점 추가 보너스\n", bonus);
                printf("보너스 포함 %4.2f점", score + bonus);
                break;
            case 7:
                printf("C 등급\n");
                bonus = 0;
                printf("추가 보너스 없음\n");
                printf("보너스 포함 %4.2f점", score + bonus);
                break;
            case 6:
                printf("D 등급\n");
                bonus = 0;
                printf("추가 보너스 없음\n");
                printf("보너스 포함 %4.2f점", score + bonus);
                break;
            default:
                printf("F 등급\n");
                bonus = 0;
                printf("추가 보너스 없음\n");
                printf("보너스 포함 %4.2f점", score + bonus);
                break;
        }
    }
}
