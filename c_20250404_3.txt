#include <stdio.h>

void main()
{
    int score = 0;
    float bonus = 0;
    char grade;
    printf("시험 점수를 입력하세요.\n");
    scanf("%d", &score);
    if (score > 100 | score < 0) {
        printf("Error");
    } else {
        if (score >= 90) {
            grade = 'A';
            printf("%c 등급\n", grade);
        } else if (score >= 80) {
            grade = 'B';
            printf("%c 등급\n", grade);
        } else if (score >= 70) {
            grade = 'C';
            printf("%c 등급\n", grade);
        } else if (score >= 60) {
            grade = 'D';
            printf("%c 등급\n", grade);
        } else {
            grade = 'F';
            printf("%c 등급\n", grade);
        }
        switch(grade)
        {
            case 'A':
                bonus = score * 0.1;
                printf("%4.2f점 추가 보너스\n", bonus);
                printf("보너스 포함 %4.2f점", score + bonus);
                break;
            case 'B':
                bonus = score * 0.05;
                printf("%4.2f점 추가 보너스\n", bonus);
                printf("보너스 포함 %4.2f점", score + bonus);
                break;
            case 'C':
                bonus = 0;
                printf("추가 보너스 없음\n");
                printf("보너스 포함 %4.2f점", score + bonus);
                break;
            case 'D':
                bonus = 0;
                printf("추가 보너스 없음\n");
                printf("보너스 포함 %4.2f점", score + bonus);
                break;
            case 'F':
                bonus = 0;
                printf("추가 보너스 없음\n");
                printf("보너스 포함 %4.2f점", score + bonus);
                break;
            default:
                printf("default");
        }
    }
}
