#include <stdio.h>

void main()
{
    int a = 0;
    scanf("%d", &a);
    if (a >= 30) {
        printf("너무 더워요! 반팔과 반바지를 입으세요.");
    } else if (a >= 25) {
        printf("따뜻한 날씨입니다! 가벼운 옷을 입으세요.");
    } else if (a >= 15) {
        printf("선선한 날씨입니다! 가벼운 겉옷을 챙기세요.");
    } else if (a >= 5) {
        printf("조금 쌀쌀합니다! 따뜻한 옷을 입으세요.");
    } else if (a >= 0) {
        printf("매우 춥습니다! 두꺼운 외투를 꼭 입으세요.");
    } else {
        printf("한파 경고! 패딩과 장갑, 모자를 착용하세요.");
    }
}
